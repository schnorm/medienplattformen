% === Zeilen- und Seitenumbruch im Blocksatz ===================================
% \hyphenpenalty leicht über dem LaTeX-Default (50). Nicht höher schrauben: ab
% etwa 500 werden Trennungen selten, dafür klaffen die Wortabstände — das
% Problem wäre nur verschoben. Gegen die auffälligen „Trennleitern" am rechten
% Rand wirken nicht hohe Penalties, sondern die beiden demerits.
\hyphenpenalty=100
\exhyphenpenalty=100
\tolerance=1500
\doublehyphendemerits=100000   % zwei Trennstriche in Folge
\finalhyphendemerits=50000     % Trennung in der vorletzten Zeile
% Notreserve, greift erst im letzten Durchgang für sonst überhängende Zeilen.
\setlength{\emergencystretch}{2em}

% Seitenumbruch. 10000 = „unendlich", der Umbruch wird damit verboten.
\clubpenalty=10000          % Schusterjunge: erste Absatzzeile allein am Seitenfuß
\widowpenalty=10000         % Hurenkind: letzte Absatzzeile allein am Seitenkopf
\displaywidowpenalty=10000  % dito vor einer abgesetzten Formel (separat nötig)
\brokenpenalty=10000        % kein getrenntes Wort über den Seitenumbruch
% Bei article+oneside ohnehin der Default — hier explizit, weil die Penalties
% oben Seiten bewusst ein bis zwei Zeilen kürzer enden lassen und \flushbottom
% den fehlenden Platz sonst aus \parskip holen würde (ungleiche Absatzabstände).
\raggedbottom

% Bindestrich, der die Silbentrennung BEIDER Wortteile erhält. Nach einem
% normalen „-" wendet TeX die Trennmuster auf den Rest des Wortes nicht mehr an
% — die häufigste Ursache für Randüberhänge im deutschen Blocksatz.
% Gemessen mit \showhyphens:
%   Betriebs-Dokumentation      -> keine einzige Trennstelle
%   Betriebs\bs Dokumentation   -> Be-triebs-Do-ku-men-ta-ti-on
% Nutzung im Fließtext: Refresh\bs Token
% (Das babel-Shorthand "= täte dasselbe und steht hier zur Verfügung, da
% shorthands NICHT abgeschaltet ist; \bs ist die explizitere Schreibweise und
% bleibt gültig, falls shorthands später doch deaktiviert wird.)
\newcommand{\bs}{\babelhyphen{hard}}
